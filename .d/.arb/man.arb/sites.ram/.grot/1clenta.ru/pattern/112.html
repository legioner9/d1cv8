Вывод текстовго документа сформированного из текстового макета
Программирование системы     03.01.2017 09:44     6994

В 1С:Предприятии печатная форма может быть выведена в текстовый документ и сформирована на основе макета, который так же может быть текстовым документом. Макет на основе текстовго документа строится по определенным правилам.



Исходный текст макета
#Область Шапка
#Поле Дата
#Выравнивание Центр
#Формат "ДФ=dd.MM.yyyy"
#Поле Наименование
#Выравнивание Лево

Поступление товаров

Номер [Номер]
Дата [Дата ]

#КонецОбласти

#Область Строка
[Наименование ]
#КонецОбласти

Заполнение и вывод текстовго документа


Процедура СформироватьТекстовыйДокумент()
	
	ТекстДок = Новый ТекстовыйДокумент;
	
	Макет = ПолучитьМакет("ТекстовыйМакет");	
	
	//Вывод области Шапка	
	ОбластьШапка = Макет.ПолучитьОбласть("Шапка");
	ОбластьШапка.Параметры.Номер = "000001";
	ОбластьШапка.Параметры.Дата = ТекущаяДата();
	ТекстДок.Вывести(ОбластьШапка);
	
	Для Индекс = 1 По 10 Цикл
	
		//Вывод области Строка
		ОбластьСтрока = Макет.ПолучитьОбласть("Строка");
		ОбластьСтрока.Параметры.Наименование = "Товар-" + Индекс;
		ТекстДок.Вывести(ОбластьСтрока);
		
	КонецЦикла; 	
	
	ТекстДок.Показать();
	
КонецПроцедуры
